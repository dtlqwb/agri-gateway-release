## 🎯 根本原因

### 主要原因：machines表未正确同步

#### 问题1: repairMachineData() 未正确执行

**代码位置**: `services/db.js` 第452-505行

```javascript
async function repairMachineData() {
  // 从 work_records 提取设备信息写入 machines 表
  const missing = await queryAll(`
    SELECT DISTINCT wr.t_number, ...
    FROM work_records wr
    WHERE wr.source = 'old'  // ❌ 只处理 source='old'
      AND wr.t_number NOT IN (SELECT t_number FROM machines WHERE source = 'old')
    GROUP BY wr.t_number
  `);
  ...
}
```

**问题**: 
- ❌ 只查询 `source = 'old'` 的记录
- ❌ 但实际数据使用的是 `source = 'old_api'`
- ❌ 导致20台设备未被同步到machines表

#### 问题2: 旧供应商数据源标识不一致

**实际情况**:
```sql
-- work_records表中的数据
SELECT DISTINCT source FROM work_records;
-- 结果: 'yuntinan', 'old_api'  (没有'old')

-- 但repairMachineData()查询的是
WHERE wr.source = 'old'  -- ❌ 匹配不到任何记录
```

**历史原因**:
- 早期使用 `source = 'old'` (Excel导入)
- 后来改为 `source = 'old_api'` (API同步)
- 但修复逻辑未更新

## ✅ 解决方案

### 方案1: 修复repairMachineData()函数（推荐）

**修改文件**: `services/db.js`

**修改内容**: 同时处理 `old` 和 `old_api` 两种数据源

```javascript
async function repairMachineData() {
  const missing = await queryAll(`
    SELECT DISTINCT wr.t_number,
           (SELECT MAX(wr2.plate_no) FROM work_records wr2 WHERE wr2.t_number = wr.t_number AND wr2.plate_no AND wr2.plate_no != '') as plate_no,
           (SELECT MAX(wr2.driver_name) FROM work_records wr2 WHERE wr2.t_number = wr.t_number AND wr2.driver_name AND wr2.driver_name != '') as driver_name,
           (SELECT MAX(wr2.org_id) FROM work_records wr2 WHERE wr2.t_number = wr.t_number AND wr2.org_id > 0) as org_id,
           (SELECT MAX(wr2.org_name) FROM work_records wr2 WHERE wr2.t_number = wr.t_number AND wr2.org_name AND wr2.org_name != '') as org_name,
           SUM(CASE WHEN COALESCE(wr.acre, 0) = 0 THEN COALESCE(wr.ok_acre, 0) ELSE wr.acre END) as total_acre
    FROM work_records wr
    WHERE wr.source IN ('old', 'old_api')  -- ✅ 修改这里
      AND wr.t_number IS NOT NULL AND wr.t_number != ''
      AND wr.t_number NOT IN (SELECT t_number FROM machines WHERE source IN ('old', 'old_api'))  -- ✅ 修改这里
    GROUP BY wr.t_number
  `);

  let count = 0;
  for (const m of missing) {
    if (!m.t_number) continue;
    await upsertMachine({
      t_number: m.t_number,
      plate_no: m.plate_no || '',
      driver_name: m.driver_name || '',
      org_id: m.org_id || 0,
      org_name: m.org_name || '',
      source: 'old',  // 统一使用'old'作为source
      year_acre: m.total_acre || 0
    });
    count++;
  }

  // 同时更新已有机器的年度面积
  const existing = await queryAll(`
    SELECT m.t_number, m.id,
           SUM(CASE WHEN COALESCE(wr.acre, 0) = 0 THEN COALESCE(wr.ok_acre, 0) ELSE wr.acre END) as total_acre
    FROM machines m
    LEFT JOIN work_records wr ON wr.t_number = m.t_number AND wr.source IN ('old', 'old_api')  -- ✅ 修改这里
    WHERE m.source IN ('old', 'old_api')  -- ✅ 修改这里
    GROUP BY m.t_number, m.id
  `);
  
  for (const m of existing) {
    await runSql(
      `UPDATE machines SET year_acre = ?, updated_at = NOW() WHERE id = ?`,
      [m.total_acre || 0, m.id]
    );
  }

  if (count > 0) {
    await refreshOrgStats();
  }

  console.log(`[DB] 修复设备数据: 新增 ${count} 台，更新 ${existing.length} 台年度面积`);
  return count;
}
```

### 方案2: 手动执行SQL修复（临时方案）

如果急需修复，可以手动执行以下SQL：

```sql
-- 1. 将缺失的20台设备插入machines表
INSERT INTO machines (t_number, plate_no, driver_name, org_id, org_name, source, year_acre)
SELECT DISTINCT 
  wr.t_number,
  MAX(wr.plate_no) as plate_no,
  MAX(wr.driver_name) as driver_name,
  MAX(wr.org_id) as org_id,
  MAX(wr.org_name) as org_name,
  'old' as source,
  SUM(CASE WHEN COALESCE(wr.acre, 0) = 0 THEN COALESCE(wr.ok_acre, 0) ELSE wr.acre END) as year_acre
FROM work_records wr
LEFT JOIN machines m ON wr.t_number = m.t_number
WHERE wr.source IN ('old', 'old_api')
  AND m.t_number IS NULL
GROUP BY wr.t_number;

-- 2. 更新所有旧供应商设备的year_acre
UPDATE machines m
INNER JOIN (
  SELECT t_number, 
         SUM(CASE WHEN COALESCE(acre, 0) = 0 THEN COALESCE(ok_acre, 0) ELSE acre END) as total_acre
  FROM work_records
  WHERE source IN ('old', 'old_api')
  GROUP BY t_number
) wr ON wr.t_number = m.t_number
SET m.year_acre = wr.total_acre,
    m.updated_at = NOW()
WHERE m.source IN ('old', 'old_api');

-- 3. 验证结果
SELECT source, COUNT(*) as device_count, SUM(year_acre) as total_acre
FROM machines
GROUP BY source;
```

async function checkDataConsistency() {
  const wrCount = await db.queryOne(
    `SELECT COUNT(DISTINCT t_number) as count FROM work_records WHERE source IN ('old', 'old_api')`

    `SELECT COUNT(*) as count FROM machines WHERE source = 'old'`
