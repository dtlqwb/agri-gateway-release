## 阶段1：稳定性加固（优先级：高）

### 任务1.1：添加全局错误处理中间件

**目标**：防止未捕获异常导致进程崩溃

**文件**：新建 `middleware/errorHandler.js`

```javascript
/**
 * 全局错误处理中间件
 */
module.exports = (err, req, res, next) => {
  console.error('[Error]', {
    message: err.message,
    stack: err.stack,
    url: req.url,
    method: req.method,
    timestamp: new Date().toISOString()
  });
  
  // 不要暴露内部错误详情给客户端
  const msg = process.env.NODE_ENV === 'production' 
    ? '服务器内部错误' 
    : err.message;
    
  res.status(500).json({ code: -1, msg });
};
```

**修改 `index.js`**：
- 在路由注册之后，添加 `app.use(require('./middleware/errorHandler'));`

### 任务1.3：数据库连接池优化

**目标**：确保连接正确释放，防止泄漏

**文件**：`services/db.js`

**修改所有查询函数**，使用 `try...finally` 模式：

```javascript
// 修改前
async function queryAll(sql, params = []) {
  const [rows] = await pool.execute(sql, params);
  return rows;
}

// 修改后
async function queryAll(sql, params = []) {
  const connection = await pool.getConnection();
  try {
    const [rows] = await connection.execute(sql, params);
    return rows;
  } finally {
    connection.release();
  }
}
```

**需要修改的函数**：`queryAll`, `queryOne`, `runSql`, `insert`, `update`

### 任务1.4：添加基础日志框架

**目标**：分级日志，便于排查问题

**文件**：新建 `utils/logger.js`

```javascript
const fs = require('fs');
const path = require('path');

// 简单日志实现（不引入外部依赖）
const LOG_DIR = path.join(__dirname, '../logs');

// 确保日志目录存在
if (!fs.existsSync(LOG_DIR)) {
  fs.mkdirSync(LOG_DIR, { recursive: true });
}

function getLogFile() {
  const date = new Date().toISOString().split('T')[0];
  return path.join(LOG_DIR, `${date}.log`);
}

function writeLog(level, message, meta = {}) {
  const timestamp = new Date().toISOString();
  const logEntry = JSON.stringify({
    timestamp,
    level,
    message,
    ...meta
  }) + '\n';
  
  fs.appendFileSync(getLogFile(), logEntry);
  
  // 控制台也输出
  console[level === 'error' ? 'error' : 'log'](`[${level.toUpperCase()}]`, message);
}

module.exports = {
  info: (msg, meta) => writeLog('info', msg, meta),
  warn: (msg, meta) => writeLog('warn', msg, meta),
  error: (msg, meta) => writeLog('error', msg, meta),
  debug: (msg, meta) => writeLog('debug', msg, meta)
};
```

**替换关键位置的 `console.log/error`**：
- 数据库错误 → `logger.error`
- API调用失败 → `logger.error`
- 同步任务状态 → `logger.info`

## 阶段2：性能优化（优先级：中）

### 任务2.1：旧供应商API并发控制

**目标**：加速同步，同时避免触发API限流

**文件**：`services/oldSupplierService.js`

**实现方式**（不引入新依赖，用原生Promise控制）：

```javascript
// 在类中添加并发控制
async syncWithConcurrency(devices, batchSize = 5) {
  const results = [];
  
  for (let i = 0; i < devices.length; i += batchSize) {
    const batch = devices.slice(i, i + batchSize);
    const batchPromises = batch.map(device => this.syncSingleDevice(device));
    const batchResults = await Promise.allSettled(batchPromises);
    
    results.push(...batchResults);
    
    // 批次间延迟，避免触发限流
    if (i + batchSize < devices.length) {
      await new Promise(r => setTimeout(r, 200));
    }
  }
  
  return results;
}
```

### 任务2.2：数据库批量写入优化

**目标**：减少数据库IO次数

**文件**：`services/db.js`

**新增批量插入函数**：

```javascript
/**
 * 批量插入作业记录（使用INSERT多值语法）
 */
async function batchInsertWorkRecords(records) {
  if (!records || records.length === 0) return 0;
  
  const connection = await pool.getConnection();
  try {
    // 构建多值INSERT
    const placeholders = records.map(() => '(?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)').join(',');
    const values = records.flatMap(r => [
      r.t_number, r.work_date, r.work_type, r.work_type_name,
      r.acre, r.ok_acre, r.repeat_acre, r.leave_acre,
      r.slng, r.slat, JSON.stringify(r.land_group || []),
      r.work_time, r.source
    ]);
    
    const sql = `
      INSERT INTO work_records 
      (t_number, work_date, work_type, work_type_name, acre, ok_acre, 
       repeat_acre, leave_acre, slng, slat, land_group, work_time, source)
      VALUES ${placeholders}
      ON DUPLICATE KEY UPDATE
      acre = VALUES(acre),
      ok_acre = VALUES(ok_acre),
      updated_at = CURRENT_TIMESTAMP
    `;
    
    const [result] = await connection.execute(sql, values);
    return result.affectedRows;
  } finally {
    connection.release();
  }
}
```

### 任务2.3：定时任务防重叠

**目标**：防止同步任务重叠执行

**文件**：`services/scheduler.js`

**添加任务锁机制**：

```javascript
class Scheduler {
  constructor() {
    this.tasks = new Map(); // 任务名称 -> 是否在运行
  }

  async runTask(name, fn) {
    if (this.tasks.get(name)) {
      console.log(`[Scheduler] 任务 ${name} 正在运行，跳过`);
      return;
    }
    
    this.tasks.set(name, true);
    try {
      await fn();
    } finally {
      this.tasks.set(name, false);
    }
  }
}
```

## 阶段3：安全与监控增强（优先级：低）

### 任务3.1：添加请求限流

**文件**：新建 `middleware/rateLimiter.js`

```javascript
// 简单内存限流（生产环境建议用Redis）
const requests = new Map();

module.exports = (windowMs = 15 * 60 * 1000, max = 100) => {
  return (req, res, next) => {
    const key = req.ip;
    const now = Date.now();
    
    if (!requests.has(key)) {
      requests.set(key, { count: 1, resetTime: now + windowMs });
      return next();
    }
    
    const record = requests.get(key);
    
    if (now > record.resetTime) {
      record.count = 1;
      record.resetTime = now + windowMs;
      return next();
    }
    
    if (record.count >= max) {
      return res.status(429).json({ code: -1, msg: '请求过于频繁，请稍后再试' });
    }
    
    record.count++;
    next();
  };
};
```
